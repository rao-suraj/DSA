class Solution {
  private:
    void solve(int i, set<string> &ans, string S, string &subs){
        // base case
        if(i >= S.size()){
            int size = subs.size();
            if(subs[0] != 'a' && subs[0] != 'e' && subs[0] != 'i' && subs[0] != 'o' && subs[0] != 'u') return;
            if(subs[size-1] == 'a' || subs[size-1] == 'e' || subs[size-1] == 'i' || subs[size-1] == 'o' || subs[size-1] == 'u') return;
            ans.insert(subs);
            return;
        }
        
        // pick
        subs.push_back(S[i]);
        solve(i+1,ans,S,subs);
        // backtrack
        subs.pop_back();
        // not pick
        solve(i+1,ans,S,subs);
    }
  public:
    set<string> allPossibleSubsequences(string S) {
        set<string> ans;
        string subs;
        solve(0,ans,S,subs);
        return ans;
    }
};

           return true;
       }
       return false;

   void findSubSeq(int index, string &S, set<string> &mini, string &output){
       if(index == S.size()){
           if(output.length() > 0){
               mini.insert(output);    
           }
           return;
       }
       // include
       output.push_back(S[index]);
       findSubSeq(index+1, S, mini, output);
       output.pop_back();
       // exclude
       findSubSeq(index+1, S, mini, output);
       return;
   }
   
   set<string> allPossibleSubsequences(string S) {
       set<string> ans;
       set<string> mini;

       int index = 0;
       // finding all possible subsequences
       findSubSeq(index, S, mini, v);
       // mini will have all possible subsequences 
       for(auto str : mini){
           int i = str.size();
           // checking given condition
           if(vowel(str[0]) && !vowel(str[i-1])){
               // checking that there is no duplicate

           }
       }
       return ans;
   }
